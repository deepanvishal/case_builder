# Research Mandate Template
# Copy per task -> work/mandates/<mandate_id>.md, fill EVERY field, hand to the hunter.
# An underspecified mandate produces garbage; the hunter REFUSES one missing required fields.

mandate_id:            # e.g. CR-rep-01
serves_claim:          # the ONE claim this must establish + criterion + reg cite
                       # e.g. "Cigna has a distinguished reputation - Critical Role, 8 CFR 204.5(h)(3)(viii)"
thesis_link:           # how it advances "petitioner is among the small % at the very top of the field"

questions:             # 2-5 CONCRETE, answerable questions (not yes/no vibes)
  -                    # e.g. "Cigna's Fortune 500 rank for 2024 and 2025?"
  -                    # e.g. "Named industry awards / analyst ratings for Cigna, 2023-2026?"

acceptance_criteria:   # what a usable answer MUST be
  - objective and quantified where possible (rank, $, %, count, date)
  - attributable to a NAMED, authoritative source
  - printable exhibit-grade (PDF with URL + access date)

source_bar:
  prefer:              # primary / authoritative only
    - company filings & investor relations; official ranking bodies (Fortune, etc.)
    - government / regulator pages; recognized standards bodies
    - the venue's / publisher's own official pages; peer-reviewed literature
    - USPTO / PPUBS for patent facts
  reject:              # discard leads from these even if they are the only hit
    - blogs, forums, Q&A sites, SEO listicles, content farms
    - Wikipedia as the cited source; crowdsourced wage/review sites
    - vendor press releases presented as independent news
    - anything undated, login-walled, or not printable as an exhibit

out_of_scope:          # explicit don'ts for THIS mandate
  -                    # e.g. "do not characterize the confidential Brighter Match deployment"
  - never infer adoption / use / endorsement from a citation or a job posting

time_bounds:           # e.g. "current standing 2024-2026" or "as of filing 2026-04-27"
search_budget:         # quick (2-3 searches) | standard (5-8) | deep (12-15)

# OUTPUT the hunter returns, per lead:
#   finding      - the literal fact, in the source's own terms (no inflation)
#   locus        - URL + page title + publisher + date + author
#   proves       - which question / claim element it satisfies
#   support_type - verified-doc | needs-source | framing-only
#   to_verify    - exact step to make it exhibit-grade
# Plus dead_ends - queries/sources checked that yielded nothing.
#
# ANTI-GARBAGE RULE: "no solid evidence found" for a question is a CORRECT, valued
# result. Never pad with weak/off-bar sources to appear productive.
